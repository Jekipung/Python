import sys
input = sys.stdin.readline

N, M = map(int, input().split())
P = []

for i in range(M):
    P.append(int(input()))

P.sort()
a_price = 0
c_price = 0

for i in range(M):
    b_price = P[i] * min(N, M - i) # 총비용 = 손님이 구매할 가격 * 구매할 갯수
    if b_price > a_price or b_price >= a_price:
        a_price = b_price
        c_price = P[i]

print(c_price, a_price)
