https://www.acmicpc.net/problem/17608

can_see = 1
stick = []
N = int(input())
for i in range(N):
    stick.append(int(input()))
IT = stick[N-1]
for i in range(N):
    if IT > stick[N-1-i]:
        continue
    elif IT == stick[N-1-i]:
        continue
    elif IT < stick[N-1-i]:
        can_see = can_see + 1
        IT = stick[N-1-i]
print(can_see)
